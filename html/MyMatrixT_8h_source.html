54  // for sorting (d first, then t, then l)
55  bool operator < (const EventTriplet& trip) const {
56  // non-pareto sort
57  //return cost < trip.cost;
58 
59  // pareto sort
60  if( d == trip.d ) {
61  if( t == trip.t )
62  return (l < trip.l);
63  else
64  return (t < trip.t);
65  }
66  return (d < trip.d);
67  }
68 
69  bool operator == (const EventTriplet& trip) const {
70  if( d == trip.d && l == trip.l && t == trip.t )
71  return true;
72  return false;
73  }
